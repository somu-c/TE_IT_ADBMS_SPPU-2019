#include <stdio.h>
#include <limits.h>

struct Process {
    int pid;
    int at;   // Arrival Time
    int bt;   // Burst Time
    int ct;   // Completion Time
    int tat;  // Turnaround Time
    int wt;   // Waiting Time
    int rt;   // Remaining Time
};

// Function to print Gantt Chart in box style
void printGanttChart(int pid[], int time[], int count) {
    int i;

    // Top border
    for (i = 0; i < count; i++) printf("+-----");
    printf("+\n");

    // Process IDs
    for (i = 0; i < count; i++) printf("| P%-2d ", pid[i]);
    printf("|\n");

    // Bottom border
    for (i = 0; i < count; i++) printf("+-----");
    printf("+\n");

    // Time labels
    for (i = 0; i <= count; i++) printf("%-6d", time[i]);
    printf("\n");
}

// ---------- FCFS Scheduling ----------
void fcfs(struct Process p[], int n) {
    printf("\n================ FCFS SCHEDULING ================\n");

    // Sort by arrival time
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (p[i].at > p[j].at) {
                struct Process tmp = p[i];
                p[i] = p[j];
                p[j] = tmp;
            }
        }
    }

    int time = 0;
    if (p[0].at > 0) time = p[0].at;

    int gcPID[50], gcTime[50];
    int gcIndex = 0;

    gcTime[0] = time;

    for (int i = 0; i < n; i++) {
        if (time < p[i].at) time = p[i].at;
        time += p[i].bt;
        p[i].ct = time;
        p[i].tat = p[i].ct - p[i].at;
        p[i].wt = p[i].tat - p[i].bt;

        gcPID[gcIndex] = p[i].pid;
        gcIndex++;
        gcTime[gcIndex] = time;
    }

    // Print Gantt chart
    printf("\nGantt Chart:\n");
    printGanttChart(gcPID, gcTime, gcIndex);

    // Results
    printf("\nPID\tAT\tBT\tCT\tTAT\tWT");
    float avgTAT = 0, avgWT = 0;
    for (int i = 0; i < n; i++) {
        printf("\nP%d\t%d\t%d\t%d\t%d\t%d", p[i].pid, p[i].at, p[i].bt, p[i].ct, p[i].tat, p[i].wt);
        avgTAT += p[i].tat;
        avgWT += p[i].wt;
    }

    printf("\nAverage Turnaround Time: %.2f", avgTAT / n);
    printf("\nAverage Waiting Time: %.2f\n", avgWT / n);
}

// ---------- SJF Preemptive (Shortest Remaining Time First) ----------
void sjf_preemptive(struct Process p[], int n) {
    printf("\n================ SJF (Preemptive) SCHEDULING ================\n");

    for (int i = 0; i < n; i++)
        p[i].rt = p[i].bt;

    int complete = 0, time = 0, min = INT_MAX, shortest = -1, check = 0;
    int prev = -1;
    int gcPID[100], gcTime[100], gcIndex = 0;

    gcTime[0] = 0;

    while (complete != n) {
        min = INT_MAX;
        shortest = -1;

        for (int j = 0; j < n; j++) {
            if (p[j].at <= time && p[j].rt > 0 && p[j].rt < min) {
                min = p[j].rt;
                shortest = j;
                check = 1;
            }
        }

        if (check == 0) {
            time++;
            continue;
        }

        if (shortest != prev) {
            gcPID[gcIndex] = p[shortest].pid;
            gcTime[gcIndex] = time;
            gcIndex++;
            prev = shortest;
        }

        p[shortest].rt--;
        time++;

        if (p[shortest].rt == 0) {
            complete++;
            p[shortest].ct = time;
            p[shortest].tat = p[shortest].ct - p[shortest].at;
            p[shortest].wt = p[shortest].tat - p[shortest].bt;
        }
    }

    gcTime[gcIndex] = time;

    // Print Gantt chart
    printf("\nGantt Chart:\n");
    printGanttChart(gcPID, gcTime, gcIndex);

    // Results
    printf("\nPID\tAT\tBT\tCT\tTAT\tWT");
    float avgTAT = 0, avgWT = 0;
    for (int i = 0; i < n; i++) {
        printf("\nP%d\t%d\t%d\t%d\t%d\t%d",
               p[i].pid, p[i].at, p[i].bt, p[i].ct, p[i].tat, p[i].wt);
        avgTAT += p[i].tat;
        avgWT += p[i].wt;
    }

    printf("\nAverage Turnaround Time: %.2f", avgTAT / n);
    printf("\nAverage Waiting Time: %.2f\n", avgWT / n);
}

// ---------- MAIN ----------
int main() {
    int n;
    printf("Enter number of processes: ");
    scanf("%d", &n);
    struct Process p[n];

    for (int i = 0; i < n; i++) {
        p[i].pid = i + 1;
        printf("Enter Arrival Time and Burst Time for Process %d: ", i + 1);
        scanf("%d %d", &p[i].at, &p[i].bt);
    }

    fcfs(p, n);
    sjf_preemptive(p, n);

    return 0;
}
